<script lang="ts" setup>
/**
 *
 * @param str “((){}[])}”
 * returns 返回是否为真
 */
const str = ref<string>("((){}[])}")
const matching = (str: string): boolean => {
  if (!str || str.length % 2 !== 0) {
    return false
  }
  let stack: string[] = []
  for (let i = 0; i < str.length; i++) {
    switch (str[i]) {
      case "(":
        stack.push("(")
        break
      case "[":
        stack.push("[")
        break
      case "{":
        stack.push("{")
        break
      case ")":
        if (stack.pop() !== "(") {
          return false
        }
        break
      case "]":
        if (stack.pop() !== "[") {
          return false
        }
        break
      case "}":
        if (stack.pop() !== "{") {
          return false
        }
        break
    }
  }
  return stack.length === 0
}
matching(str.value)
</script>
